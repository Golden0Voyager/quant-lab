import akshare as ak

def fetch_operating_cashflow_direct(symbol: str) -> float:
    """
    直接获取经营现金流总额（最近报告期）

    数据源: AKShare - 个股现金流量表
    """
    try:
        # 获取现金流量表
        df = ak.stock_financial_analysis_indicator(symbol=symbol)

        # 筛选最近报告期
        latest = df.iloc[0]

        # 经营活动现金流净额（单位：元）
        operating_cf = latest.get('经营活动产生的现金流量净额', None)

        if operating_cf and operating_cf > 0:
            return float(operating_cf)

    except Exception as e:
        logger.warning(f"直接获取现金流失败: {e}")

    return None


def calculate_pcf_v2(symbol: str, market_cap: float) -> float:
    """
    优化版PCF计算（直接数据源）

    PCF = 总市值 / 经营现金流总额

    operating_cf = fetch_operating_cashflow_direct(symbol)

def calculate_ps_pcf(self, symbol: str, stock_name: str, market_cap: float) -> Dict[str, Any]:
    """优化版：多数据源 + 降级策略"""

    data = {'ps_ttm': None, 'pcf': None, 'fcf': None}

    # 1. 计算PS（保持不变）
    data['ps_ttm'] = self._calculate_ps(symbol, stock_name, market_cap)

    # 2. 计算PCF（新增多数据源）
    # 优先级1: 直接获取经营现金流
    operating_cf = self._fetch_operating_cf_direct(symbol)

    if operating_cf and operating_cf > 0:
        data['pcf'] = market_cap / operating_cf
        logger.info(f"✅ PCF（直接数据）: {data['pcf']:.2f}")
    else:
        # 优先级2: 从东财财务摘要获取
        eastmoney_cf = self._fetch_cashflow_from_eastmoney(symbol)
        if eastmoney_cf.get('operating_cf'):
            data['pcf'] = market_cap / eastmoney_cf['operating_cf']
            logger.info(f"✅ PCF（东财数据）: {data['pcf']:.2f}")
        else:
            # 优先级3: 原逻辑（反推）
            data['pcf'] = self._calculate_pcf_fallback(symbol, market_cap)
            if data['pcf']:
                logger.info(f"✅ PCF（反推逻辑）: {data['pcf']:.2f}")
            else:
                logger.warning("⚠️  PCF计算失败（所有方法）")

    # 3. 计算自由现金流（新增）
    free_cf = self._calculate_free_cashflow(symbol)
    if free_cf and free_cf > 0:
        data['p_fcf'] = market_cap / free_cf  # 市值/自由现金流

    return data

### 1.3 优化历史分位计算

#### 优化方案：自建历史分位计算

```python
def calculate_percentile_self(symbol: str, indicator: str = 'pe') -> dict:
    """
    自建历史分位计算（基于AKShare历史数据）

    Args:
        symbol: 股票代码
        indicator: 指标类型 (pe/pb/ps)

    Returns:
        {
            '10y': 25.5,  # 当前值在10年历史中的分位数
            '5y': 15.2,
            '3y': 10.8,
            '1y': 5.3,
            'current': 18.5,  # 当前值
            'min_10y': 12.3,  # 10年最小值
            'max_10y': 85.2,  # 10年最大值
            'mean_10y': 32.5, # 10年均值
            'median_10y': 28.3, # 10年中位数
        }
    """
    import akshare as ak
    import pandas as pd
    import numpy as np
    from datetime import datetime, timedelta

    # 获取历史估值数据（需要日级数据）
    # 方法1: 使用东方财富历史估值
    try:
        df = ak.stock_zh_valuation_baidu(
            symbol=symbol,
            indicator='市盈率(TTM)',
            period='近十年'
        )

        # 计算分位数
        current_value = df['value'].iloc[-1]

        result = {'current': current_value}

        # 不同时间窗口
        windows = {
            '10y': len(df),
            '5y': min(252 * 5, len(df)),  # 5年交易日
            '3y': min(252 * 3, len(df)),
            '1y': min(252, len(df)),
        }

        for period, days in windows.items():
            if days <= 0:
                continue

            recent_data = df.tail(days)['value']

            # 计算分位数
            percentile = (recent_data < current_value).sum() / len(recent_data) * 100
            result[period] = round(percentile, 1)

            # 额外统计量
            result[f'min_{period}'] = recent_data.min()
            result[f'max_{period}'] = recent_data.max()
            result[f'mean_{period}'] = recent_data.mean()
            result[f'median_{period}'] = recent_data.median()

        return result

    except Exception as e:
        logger.error(f"自建分位计算失败: {e}")
        return {}
```

```python
class DataValidator:
    """数据质量验证器"""

    @staticmethod
    def validate_valuation_metrics(metrics: ValuationMetrics) -> list:
        """
        验证估值指标的合理性

        Returns:
            warnings: 警告列表
        """
        warnings = []

        # 1. PE合理性检查
        if metrics.pe_ttm:
            if metrics.pe_ttm < 0:
                warnings.append("⚠️  PE为负，公司可能处于亏损状态")
            elif metrics.pe_ttm > 200:
                warnings.append("⚠️  PE过高(>200)，估值可能失真或成长性极强")

        # 2. PB合理性检查
        if metrics.pb:
            if metrics.pb < 0.5:
                warnings.append("⚠️  PB < 0.5，可能资产质量存疑或严重低估")
            elif metrics.pb > 20:
                warnings.append("⚠️  PB > 20，轻资产公司或估值过高")

        # 3. PS合理性检查
        if metrics.ps_ttm:
            if metrics.ps_ttm > 50:
                warnings.append("⚠️  PS > 50，营收规模较小或估值泡沫")

        # 4. 分位数一致性检查
        if metrics.pe_percentile_10y and metrics.pb_percentile_10y:
            diff = abs(metrics.pe_percentile_10y - metrics.pb_percentile_10y)
            if diff > 50:
                warnings.append(
                    f"⚠️  PE分位({metrics.pe_percentile_10y:.1f}%)与"
                    f"PB分位({metrics.pb_percentile_10y:.1f}%)差异较大，"
                    "可能公司业务模式或资产结构发生重大变化"
                )

        # 5. 股息率合理性
        if metrics.dividend_yield:
            if metrics.dividend_yield > 10:
                warnings.append("⚠️  股息率>10%，可能是特殊分红或股价暴跌")

        # 6. PCF缺失警告
        if not metrics.pcf:
            warnings.append("ℹ️  PCF数据缺失，现金流分析不完整")

        return warnings
```

```python
class BacktestEngine:
    """
    简单回测引擎

    功能：
    1. 记录历史分析建议
    2. 跟踪后续股价表现
    3. 统计胜率和收益率
    """

    def __init__(self, db_path="backtest.db"):
        self.db_path = db_path
        self._init_db()

    def _init_db(self):
        """初始化数据库"""
        import sqlite3
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()

        # 建议记录表
        cursor.execute("""
        CREATE TABLE IF NOT EXISTS recommendations (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            date TEXT NOT NULL,
            symbol TEXT NOT NULL,
            stock_name TEXT,
            analysis_mode TEXT,
            prompt_version TEXT,
            rating TEXT,  -- 看多/中性/看空
            confidence REAL,
            entry_price REAL,
            target_price REAL,
            stop_loss REAL,
            signal_score INTEGER,
            triggers TEXT,
            created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        )
        """)

        # 跟踪记录表
        cursor.execute("""
        CREATE TABLE IF NOT EXISTS tracking (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            recommendation_id INTEGER,
            check_date TEXT,
            current_price REAL,
            return_pct REAL,
            status TEXT,  -- holding/hit_target/hit_stop/expired
            FOREIGN KEY (recommendation_id) REFERENCES recommendations(id)
        )
        """)

        conn.commit()
        conn.close()

    def log_recommendation(self, data: dict):
        """
        记录分析建议

        Args:
            data: {
                'symbol': '600519',
                'stock_name': '贵州茅台',
                'rating': '看多',
                'confidence': 75,
                'entry_price': 1580,
                'target_price': 1750,
                'stop_loss': 1480,
                ...
            }
        """
        import sqlite3
        from datetime import datetime

        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()

        cursor.execute("""
        INSERT INTO recommendations
        (date, symbol, stock_name, analysis_mode, prompt_version,
         rating, confidence, entry_price, target_price, stop_loss,
         signal_score, triggers)
        VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
        """, (
            datetime.now().strftime('%Y-%m-%d'),
            data['symbol'],
            data.get('stock_name'),
            data.get('analysis_mode'),
            data.get('prompt_version'),
            data.get('rating'),
            data.get('confidence'),
            data.get('entry_price'),
            data.get('target_price'),
            data.get('stop_loss'),
            data.get('signal_score'),
            ','.join(data.get('triggers', []))
        ))

        conn.commit()
        rec_id = cursor.lastrowid
        conn.close()

        logger.info(f"📝 已记录建议 #{rec_id}: {data['symbol']} - {data.get('rating')}")
        return rec_id

    def update_tracking(self):
        """
        批量更新所有未完成建议的跟踪状态
        """
        import akshare as ak
        import sqlite3
        from datetime import datetime

        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()

        # 获取所有未完成的建议
        cursor.execute("""
        SELECT id, symbol, entry_price, target_price, stop_loss, date
        FROM recommendations
        WHERE id NOT IN (
            SELECT DISTINCT recommendation_id
            FROM tracking
            WHERE status IN ('hit_target', 'hit_stop', 'expired')
        )
        """)

        recommendations = cursor.fetchall()

        for rec in recommendations:
            rec_id, symbol, entry, target, stop, rec_date = rec

            try:
                # 获取当前价格
                df = ak.stock_zh_a_spot_em()
                row = df[df['代码'] == symbol].iloc[0]
                current_price = float(row['最新价'])

                # 计算收益率
                return_pct = (current_price - entry) / entry * 100 if entry else 0

                # 判断状态
                if target and current_price >= target:
                    status = 'hit_target'
                elif stop and current_price <= stop:
                    status = 'hit_stop'
                elif (datetime.now() - datetime.strptime(rec_date, '%Y-%m-%d')).days > 180:
                    status = 'expired'
                else:
                    status = 'holding'

                # 记录跟踪
                cursor.execute("""
                INSERT INTO tracking (recommendation_id, check_date, current_price, return_pct, status)
                VALUES (?, ?, ?, ?, ?)
                """, (rec_id, datetime.now().strftime('%Y-%m-%d'), current_price, return_pct, status))

                logger.info(f"✅ 更新跟踪 #{rec_id}: {symbol} {return_pct:+.2f}% ({status})")

            except Exception as e:
                logger.error(f"❌ 更新失败 #{rec_id}: {e}")

        conn.commit()
        conn.close()

    def get_performance_report(self, days=30) -> dict:
        """
        生成回测表现报告

        Args:
            days: 统计最近N天的建议

        Returns:
            {
                'total_count': 50,
                'win_rate': 62.5,
                'avg_return': 8.3,
                'best_trade': {...},
                'worst_trade': {...},
                'by_rating': {...},
                'by_prompt_version': {...},
            }
        """
        import sqlite3
        import pandas as pd
        from datetime import datetime, timedelta

        conn = sqlite3.connect(self.db_path)

        # SQL查询
        query = f"""
        SELECT
            r.*,
            t.current_price,
            t.return_pct,
            t.status
        FROM recommendations r
        LEFT JOIN (
            SELECT recommendation_id, current_price, return_pct, status
            FROM tracking
            WHERE id IN (
                SELECT MAX(id) FROM tracking GROUP BY recommendation_id
            )
        ) t ON r.id = t.recommendation_id
        WHERE r.date >= date('now', '-{days} days')
        """

        df = pd.read_sql_query(query, conn)
        conn.close()

        # 统计分析
        completed = df[df['status'].isin(['hit_target', 'hit_stop', 'expired'])]

        report = {
            'total_count': len(df),
            'completed_count': len(completed),
            'win_count': len(completed[completed['return_pct'] > 0]),
            'win_rate': (completed['return_pct'] > 0).mean() * 100 if len(completed) > 0 else 0,
            'avg_return': completed['return_pct'].mean() if len(completed) > 0 else 0,
            'median_return': completed['return_pct'].median() if len(completed) > 0 else 0,
            'best_trade': completed.nlargest(1, 'return_pct').to_dict('records')[0] if len(completed) > 0 else None,
            'worst_trade': completed.nsmallest(1, 'return_pct').to_dict('records')[0] if len(completed) > 0 else None,
        }

        # 按评级统计
        report['by_rating'] = completed.groupby('rating')['return_pct'].agg(['count', 'mean', 'median']).to_dict()

        # 按Prompt版本统计
        report['by_prompt_version'] = completed.groupby('prompt_version')['return_pct'].agg(['count', 'mean', 'median']).to_dict()

        return report
```

```python
class RiskMonitor:
    """
    风险监控与预警

    功能:
    1. 持仓股票异常波动预警
    2. 财报地雷检测
    3. 技术面破位预警
    4. 估值泡沫警报
    """

    def check_price_alert(self, watchlist: list) -> list:
        """
        价格异动预警

        触发条件:
        - 单日涨跌>7%
        - 3日累计涨跌>15%
        - 突破关键技术位
        """
        alerts = []

        for stock in watchlist:
            # 获取最近价格
            df = fetch_recent_prices(stock['code'], days=3)

            # 单日异动
            latest_change = df['涨跌幅'].iloc[-1]
            if abs(latest_change) > 7:
                alerts.append({
                    'symbol': stock['code'],
                    'type': 'price_spike',
                    'severity': 'high',
                    'message': f"单日{'大涨' if latest_change > 0 else '大跌'}{abs(latest_change):.2f}%",
                })

            # 3日累计
            cumulative_change = (df['收盘'].iloc[-1] / df['收盘'].iloc[0] - 1) * 100
            if abs(cumulative_change) > 15:
                alerts.append({
                    'symbol': stock['code'],
                    'type': 'trend_alert',
                    'severity': 'medium',
                    'message': f"3日累计{'上涨' if cumulative_change > 0 else '下跌'}{abs(cumulative_change):.2f}%",
                })

        return alerts

    def check_fundamental_risk(self, symbol: str) -> list:
        """
        基本面风险检测

        地雷检测:
        - 应收账款/营收 > 50%
        - 商誉/净资产 > 30%
        - 资产负债率 > 70%
        - 经营现金流/净利润 < 0.5
        """
        risks = []

        # 获取财务数据
        financials = fetch_financial_indicators(symbol)

        # 应收账款风险
        if financials.get('应收账款占比') > 50:
            risks.append({
                'type': 'receivable_risk',
                'severity': 'high',
                'message': f"应收账款占营收{financials['应收账款占比']:.1f}%，存在坏账风险",
            })

        # 商誉风险
        if financials.get('商誉占比') > 30:
            risks.append({
                'type': 'goodwill_risk',
                'severity': 'medium',
                'message': f"商誉占净资产{financials['商誉占比']:.1f}%，存在减值风险",
            })

        # 负债风险
        if financials.get('资产负债率') > 70:
            risks.append({
                'type': 'leverage_risk',
                'severity': 'medium',
                'message': f"资产负债率{financials['资产负债率']:.1f}%，杠杆较高",
            })

        # 现金流风险
        cf_quality = financials.get('经营现金流') / financials.get('净利润')
        if cf_quality < 0.5:
            risks.append({
                'type': 'cashflow_risk',
                'severity': 'high',
                'message': f"经营现金流/净利润仅{cf_quality:.2f}，利润质量差",
            })

        return risks
```

class AlertManager:
    """
    预警管理器

    支持:
    - 企业微信Webhook
    - 邮件通知
    - Telegram Bot
    - 钉钉机器人
    """

    def send_wechat_alert(self, message: str):
        """企业微信推送"""
        import requests

        webhook_url = os.getenv("WECHAT_WEBHOOK_URL")

        payload = {
            "msgtype": "markdown",
            "markdown": {
                "content": message
            }
        }

        requests.post(webhook_url, json=payload)

    def send_email_alert(self, subject: str, content: str):
        """邮件通知"""
        import smtplib
        from email.mime.text import MIMEText

        # SMTP配置
        smtp_server = os.getenv("SMTP_SERVER")
        smtp_port = int(os.getenv("SMTP_PORT", 465))
        sender = os.getenv("EMAIL_SENDER")
        password = os.getenv("EMAIL_PASSWORD")
        recipient = os.getenv("EMAIL_RECIPIENT")

        # 发送邮件
        msg = MIMEText(content, 'html', 'utf-8')
        msg['Subject'] = subject
        msg['From'] = sender
        msg['To'] = recipient

        with smtplib.SMTP_SSL(smtp_server, smtp_port) as server:
            server.login(sender, password)
            server.sendmail(sender, [recipient], msg.as_string())

def _fetch_operating_cf_direct(self, symbol: str) -> Optional[float]:
    """
    直接获取经营现金流总额（优先方法）

    Returns:
        经营现金流（元），失败返回None
    """
    try:
        # 方法1: 主要财务指标
        df = ak.stock_financial_abstract_ths(symbol=symbol, indicator="按报告期")

        if not df.empty:
            # 获取最近报告期
            latest = df.iloc[0]

            # 每股经营现金流（元）
            cf_per_share = latest.get('每股经营现金流')

            # 总股本（万股）
            # 需要从股本结构获取
            capital_df = ak.stock_share_change_cninfo(symbol=symbol)
            if not capital_df.empty:
                total_shares = capital_df.iloc[0]['总股本'] * 10000  # 万股 -> 股

                if cf_per_share and total_shares:
                    operating_cf = cf_per_share * total_shares
                    logger.info(f"✅ 经营现金流（直接）: {operating_cf/1e8:.2f}亿")
                    return operating_cf
