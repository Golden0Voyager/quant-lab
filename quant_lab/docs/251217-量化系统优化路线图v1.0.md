def calculate_relevance_score(news_title, stock_name, stock_code):
    score = 0
    if stock_name in news_title:
        score += 10
    if stock_code in news_title:
        score += 10
    # 关键词权重
    keywords = ['业绩', '公告', '分红', '重组', '增持']
    for kw in keywords:
        if kw in news_title:
            score += 5
    return score

# 只保留高分新闻
news_list = [n for n in news_list if calculate_relevance_score(n, ...) > 15]

    for stock in watchlist:

        if sector not in sectors:
            sectors[sector] = []
        sectors[sector].append(stock)

    # 统计各板块资金流向

def backtest_accuracy():
    predictions = db.get_last_week()
    correct = 0
    for p in predictions:
        actual_change = get_price_change(p['code'], p['date'], days=7)
        if (p['prediction'] == '看多' and actual_change > 0) or \
           (p['prediction'] == '看空' and actual_change < 0):

    accuracy = correct / len(predictions) * 100
